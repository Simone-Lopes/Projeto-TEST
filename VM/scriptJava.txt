#!/bin/bash

java -version #verifica a versao do java(se existir)

if [ $? = 0 ];
then
        echo "Java instalado!"
        else
        echo "Java nao instalado..."
        echo "Gostaria de baixar o Java? [s/n]"
        read get
if [ "$get" == "s" ];
then
        sudo apt install openjdk-17-jre -y
        fi

fi
